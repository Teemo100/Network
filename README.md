# 
How to use:
  Add it in your root build.gradle at the end of repositories:
  allprojects {
		repositories {
			...
			maven { url 'https://jitpack.io' }
		}
	}
  
  Add the dependency
  dependencies {
	        implementation 'com.github.Teemo100:Network:1.0.0'
	}
